class A(object):
  def __init__(self):
    for attr,v in self.ATTRS.iteritems():
      setattr(self,attr,v)
    print "A Created"
    self._a = 'a'
  @property
  def a(self):
    return self._a

class B(A):
  ATTRS = {'v':['v','w']}
  def __init__(self, c):
    super(B, self).__init__()
    self._b = 'b'
    self._c = c
  @property
  def b(self):
    return self._b
  @property
  def c(self):
    return self._c
